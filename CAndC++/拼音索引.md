const char *PinYin_Index(const char *pPinyin)
{
    PY_Index *cpHZ,*cpHZedge; 
    INT8U PYBuff[10], i, PYLen;
    
    if(NULL == pPinyin){        /* 指针空 */
        return NULL;
    }
 
    PYLen=strlen(pPinyin);    /*输入拼音串长度*/ 
    if(0 == PYLen){        /*如果输入空字符返回0*/ 
        return NULL;
    }
    else if(sizeof(PYBuff) < PYLen){
        PYLen = sizeof(PYBuff);
    }
 
    for(i=0; i<PYLen; i++){
        PYBuff[i] = pPinyin[i] | 0x20;             /*将字母串转为小写*/ 
    }

    if(('a' > PYBuff[0])               /*错误拼音输入*/ 
        || ('z' < PYBuff[0])
        || ('i' == PYBuff[0]) 
        || ('u' == PYBuff[0]) 
        || ('v' == PYBuff[0])){
        return NULL;
    }
 
    cpHZ=PY_index_headletter[PYBuff[0]-'a'];        /*查首字母索引*/ 
    cpHZedge=PY_index_headletter[PYBuff[0]-'a'+1];  /*设置指针界限*/ 
 
    while(cpHZ < cpHZedge)                         /*索引表不超界*/ 
    { 
        for(i=0; i<PYLen; i++)                  /* 从第二个字母开始比较，包括最后的'\0' */
        { 
            if(PYBuff[i+1] != (*cpHZ).PY[i])   /*发现字母串不配,退出*/
            {
                break;
            }  
        } 
        if(i == PYLen)        /*字母串全配*/ 
        { 
            return (*cpHZ).PY_mb; 
        } 
        cpHZ++; 
    } 
    
    return NULL;                      /*无果而终*/ 
}
