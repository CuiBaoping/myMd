# 可变参数

~~~c
#include <stdarg.h>
int sum(int num_args, ...)
{
   int val = 0;
   va_list ap;		//定义va_list类型（指针）
   int i;

   va_start(ap, num_args);		//取得函数参数的开始位置
   for(i = 0; i < num_args; i++)//num_args指出可变参数的个数
   {
      val += va_arg(ap, int);	//获取一个参数，并将位置指向下一个参数
   }
   va_end(ap);					//释放va_list类型（指针）
 
   return val;
}
~~~
